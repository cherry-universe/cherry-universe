# 在网页上下载网易云音乐的音乐

a.py

```python
import webbrowser;
import requests;

def download(url):
    """下载音乐源文件"""
    res=requests.get(url);
    if (res.status_code==requests.codes.ok):
        print("Successfully downloaded "+url);
    else:
        print("Doenload faild.");
        exit(1);
    return res;

def writeToMp3(res,name):
    """写入本地mp3文件"""
    file=open(name+".mp3","wb");
    for i in res.iter_content(100000):
        file.write(i);
    file.close();

class MusicMessage:
    """音乐信息"""
    def __init__(self):
        self.id=0;
        self.name="";
        self.url="";

#获取音乐信息
musicMessage=MusicMessage();
print("The music's id: ",end='');
musicMessage.id=input();
print("Name: ",end='');
musicMessage.name=input();

#拼接url字符串
musicMessage.url="http://music.163.com/song/media/outer/url?id=";
musicMessage.url+=str(musicMessage.id);
musicMessage.url+=".mp3";

#打开浏览器
print("URL:"+musicMessage.url);
print("Opening webbrowser...");
webbrowser.open(musicMessage.url);

#下载音乐
print("Downloading music...");
res=download(musicMessage.url);
writeToMp3(res,musicMessage.name);

print("Done.");
```
